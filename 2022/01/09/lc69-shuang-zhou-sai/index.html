        vector<string>h;
        string s="";
        for(auto& i:title){

                }

            len--;
            if(len>0)
                s+=" ";
        }
        return s;
    }
};

class Solution {

    int pairSum(ListNode* head) {
        vector<int>ha;
        while(head){
            ha.emplace_back(head->val);
            head=head->next;
        }
        int ma=INT_MIN,n=ha.size()/2-1,len=ha.size();
        for(int i=0;i<=n;++i){
            ma=max(ma,ha[i]+ha[len-i-1]);
